
DATA DICTIONNARY

Sounds :
id - SERIAL PK
sound_name - TEXT UNIQUE NOT NULL (Displayed name)
audio_paths - TEXT[] NOT NULL (Audio paths for a sound, a sound can play a random audio from these paths)
image_path - TEXT
looping - BOOLEAN (Make a sound loop)
volume - INT (Set volume)
reverb - INT (Set reverb)
reverb_duration - DECIMAL
speed - DECIMAL (Playback speed)
direction - INT (Set left and right panning)
category - ENUM(TEXT) (Nature, Animals, Human...)
themes - ENUM[](TEXT) (Fantasy, Aquatic, Mysterious...)
repeat_delay - DECIMAL[] (Set minimum and maximum delay between played audios)

Ambiances :
id - SERIAL PK
ambiance_name - TEXT NOT NULL
author_id - TEXT (The user who created the ambiance)
categories - ENUM[](TEXT) (Nature, Animals, Human...)
themes - ENUM[](TEXT) (Fantasy, Aquatic, Mysterious...)

Ambiances_Sounds :
id - SERIAL PK,
ambiance_id - INT NOT NULL (One ambiance has multiple sounds)
sound_id - INT NOT NULL
volume - INT,
direction - DECIMAL
speed - DECIMAL
reverb - INT
reverb_duration - DECIMAL
low - INT (Low frequencies volume of the sound)
mid - INT (Mid frequencies volume of the sound)
high - INT (High frequencies volume of the sound)
low_cut - INT (Remove low frequencies of the sound below a chosen frequency)
high_cut - INT (Remove high frequencies of the sound after a chosen frequency)
repeat_delay - DECIMAL[]

User_has_favorite_sounds :
id - SERIAL PK
user_id - TEXT NOT NULL
sound_id - INT NOT NULL

User_has_favorite_ambiances :
id - SERIAL PK
user_id - TEXT NOT NULL
ambiance_id - INT NOT NULL

Ips_download_limits :
id - SERIAL PK
ip_address - TEXT NOT NULL UNIQUE
number_of_sounds_downloaded - INT NOT NULL
reset_date - TIMESTAMP WITH TIME ZONE NOT NULL

User :
id - TEXT PK NOT NULL
name - TEXT NOT NULL
email - TEXT UNIQUE NOT NULL
emailVerified - BOOLEAN NOT NULL
image - TEXT
createdAt - TIMESTAMP NOT NULL
updatedAt - TIMESTAMP NOT NULL
role - TEXT NOT NULL

Session :
id - TEXT PK NOT NULL
expiresAt - TIMESTAMP NOT NULL
token - TEXT UNIQUE NOT NULL
createdAt - TIMESTAMP NOT NULL
updatedAt - TIMESTAMP NOT NULL
ipAddress - TEXT
userAgent - TEXT
userId - TEXT NOT NULL FK(user.id)

Account :
id - TEXT PK NOT NULL
accountId - TEXT NOT NULL
providerId - TEXT NOT NULL
userId - TEXT NOT NULL FK(user.id)
accessToken - TEXT
refreshToken - TEXT
idToken - TEXT
accessTokenExpiresAt - TIMESTAMP
refreshTokenExpiresAt - TIMESTAMP
scope - TEXT
password - TEXT
createdAt - TIMESTAMP NOT NULL
updatedAt - TIMESTAMP NOT NULL

Verification :
id - TEXT PK NOT NULL
identifier - TEXT NOT NULL
value - TEXT NOT NULL
expiresAt - TIMESTAMP NOT NULL
createdAt - TIMESTAMP
updatedAt - TIMESTAMP
